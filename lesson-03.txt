// Bài tập git-1
git commit --amend : undo tất cả các message
git commit --amend -m”<message>” : undo message chỉ định sai
git restore --staged <file> : undo các file từ staging -> working directory
git reset HEAD~1 : undo 1 commit (chuyển từ file repo -> working directory)
